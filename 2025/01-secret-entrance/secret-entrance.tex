\documentclass{article}
\usepackage{ifthen}
\usepackage{tikz}
\newif\ifcontinue
\makeatletter
\newcount\parselinenumber
\def\parseline#1{%
    \expandafter\parseline@i#1\relax
}
\def\parseline@i#1#2\relax{%
    \def\firstchar{#1}%
    \edef\numberpart{#2}%
    \parselinenumber=#2\relax
}


\begin{document}

% Open the file
\newread\myfile
% \openin\myfile=example.txt
\openin\myfile=input.txt

% Read it
\newtoks\content

\newcount\position
\position=50
\newcount\onzero
\onzero=0
\newcount\overzero
\overzero=0

\continuetrue
\loop
\read\myfile to \content

\ifeof\myfile
    \continuefalse
\else
    \parseline{\content}
    % I have read \content. The chart is \firstchar, and the number is \numberpart.
    \def\Lchar{L}%
    \ifx\firstchar\Lchar
        % I avance by \numberpart.
        \advance \position by -\numberpart
    \else
        % I recule by \numberpart.
        \advance \position by \numberpart
    \fi
    \pgfmathsetcount{\position}{Mod(\the\position,100)}
    % Position is \the\position
    \ifnum\position=0
        \advance\onzero by 1
    \fi


\fi
\ifcontinue
\repeat

% Finally close the file
\closein\myfile

Part I : \the\onzero.

\end{document}
